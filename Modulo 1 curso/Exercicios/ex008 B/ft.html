    a = int(input('digite um numero: '))
    b = int(input('digite outro numero: '))
    
    def soma(a, b):
        return a + b
    def sub(a, b):
        return a - b
    def div(a, b):
        return a / b
    def multi(a, b):
        return a * b
    
    print('a divisão entre os numeros é: ', div(a,b))
    print('a subtração entre os numeros é: ', sub(a,b))
    print('a multiplicação entre os numeros é: ' ,multi(a,b))
    print('a soma entre os numeros é: ', soma(a,b))
